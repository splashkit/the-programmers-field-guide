Watch this video to:

- Get started using the while loop.

At the end of this video you should be able to write simple while loops within your programs.

```c++ {5-20}
#include "splashkit.h"

int main()
{
    int x, y;
    open_window("Drawing!", 1000, 720);

    clear_screen(COLOR_BLACK);

    while(!quit_requested())
    {
        process_events();
        if (mouse_down(LEFT_BUTTON))
        {
            x = mouse_x();
            y = mouse_y();
            fill_circle(COLOR_WHITE, x, y, 10);
        }
        refresh_screen();
    }

}
```
